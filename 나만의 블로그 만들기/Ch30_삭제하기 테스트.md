```java
	//삭제하기
	@DeleteMapping("/dummy/user/{id}")
	public String delete(@PathVariable int id) {
		try {
			userRepositoy.deleteById(id);
		} catch (EmptyResultDataAccessException e) {
			return "삭제에 실패했습니다.";
		}
		return "삭제되었습니다.";
	}
```
